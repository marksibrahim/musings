\begin{align}
x &= y \newline
x^2 &= xy  \newline
x^2 -y^2 &= xy -y^2 \newline
(x+y)(x-y) &= y(x-y) \newline
x+y &= y \newline
y + y &= y \newline
2y &= y \newline
2 &= 1
\end{align}
